patient_medicine_doctor
create table patient_medicine(P_Id int,D_Id int,M_Id int,qty int,DoI varchar(32),PRIMARY KEY(P_Id,D_Id,M_Id),FOREIGN KEY(P_Id) REFERENCES patient(P_Id),FOREIGN KEY(D_Id) REFERENCES doctor(D_Id),FOREIGN KEY(P_Id) REFERENCES medicines(M_Id));
